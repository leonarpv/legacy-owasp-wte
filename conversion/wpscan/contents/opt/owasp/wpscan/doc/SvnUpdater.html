def is_installed?
  %[svn info "#@repo_directory" --xml 2>&1] =~ /revision=/ ? true : false
end

def local_revision_number
  local_revision = %[svn info "#@repo_directory" --xml 2>&1]
  local_revision[REVISION_PATTERN, 1].to_s
end
